
使用haar特征分类器训练训练得到的人脸检测模型
haarcascade_frontalcatface.xml
```c++
#include <opencv2/opencv.hpp>
#include <opencv2/imgproc/imgproc.hpp>
#include <opencv2/highgui/highgui.hpp>
#include <opencv2/objdetect/objdetect.hpp>
#include <iostream>
#include <vector>

int main()
{
    cv::CascadeClassifier face_cascade;
    if (!face_cascade.load("haarcascade_frontalcatface.xml"))
    {
        std::cerr << "Error loading face cascade file." << std::endl;
        return -1;
    }
    cv::Mat image = cv::imread("xxx.jpg");
    if (image.empty())
    {
        std::cerr << "Error loading image." << std::endl;
        return -1;
    }
    std::vector<cv::Rect> faces;
    face_cascade.detectMultiScale(image, faces);
    for (size_t i = 0; i < faces.size(); i++)
    {
        cv::rectangle(image, faces[i], cv::Scalar(255, 0, 0), 2);
    }
    cv::imshow("Face Detection", image);
    cv::waitKey(0); // 等待按键后关闭窗口
    return 0;
}
```
